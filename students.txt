1. SELECT s.name
   FROM students s
   JOIN class c
   ON c.student_id = s.student_id
   WHERE c.class_id = 101;
   
2. SELECT COUNT(grade)
   FROM grades
   WHERE class_id = 999
   AND grade = 'B';
   
3. SELECT name, COUNT(student_id)
   FROM class
   GROUP BY name;
   
4. SELECT name, COUNT(student_id)
   FROM class
   GROUP BY name
   ORDER BY COUNT(student_id) DESC
   LIMIT 1;
