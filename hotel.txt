1. SELECT g.name
   FROM guest g
   LEFT JOIN booking b ON b.guest_id = g.guest_id
   WHERE b.confirmation_number IS NULL;

2. SELECT g.name, b.room_number
   FROM guest g
   JOIN booking b ON b.guest_id = g.guest_id
   GROUP BY b.check_in
   HAVING COUNT(*) > 1;

3. SELECT g.name, b.room_number
   FROM guest g
   JOIN booking b ON b.guest_id = g.guest_id
   GROUP BY b.room_number
   HAVING COUNT(*) > 1;
   
4. SELECT DISTINCT g.name, b.room_number
   FROM guest g
   JOIN booking b ON b.guest_id = g.guest_id
   GROUP BY b.room_number
   HAVING COUNT(*) > 1;
